# list topics
docker exec broker kafka-topics --bootstrap-server broker:9092 --list

# create topic
docker exec broker kafka-topics --bootstrap-server broker:9092 --create --topic <topic_name>

# show topic messages
docker exec broker kafka-console-consumer --bootstrap-server broker:9092 --topic <topic_name> --from-beginning

# describe topics
docker exec broker kafka-topics --bootstrap-server broker:9092 --describe

# partitioning topic
docker exec broker kafka-topics --alter --bootstrap-server broker:9092 --topic <topic_name> --partitions <number_of_partitions>
